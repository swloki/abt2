销售模块 (CRM) — 类图设计

5 个核心实体 + 共享层引用 + 完整业务流转

Sales Module — UML Class Diagram

业务流转

Quotation 报价 ──DERIVED_FROM──→ SalesOrder 销售订单 ──TRIGGERS──→ ShippingRequest 发货申请 ──RECONCILES──→ Reconciliation 月对账单
↗ SalesReturn 销售退货 （从 ShippingRequest 逆向触发）

Quotation 报价单

功能：结合 BOM 成本规范商务询报价，锁定订单初始利润

entity Quotation {
  id: UUID PK
  doc_number: String // QUO-2026-05-xxxxx
  customer_id: UUID FK→customers
  contact_id: UUID FK→contacts
  quotation_date: Date
  valid_until: Date // 报价有效期
  status: QuotationStatus
  total_amount: Decimal(10,6)
  total_cost: Decimal(10,6)
  estimated_margin: Decimal(5,2) // 预估毛利率%
  payment_terms: String // 付款条件
  delivery_terms: String // 交货条款
  remark: String
  operator_id: UUID FK→users
  created_at: Timestamp
  updated_at: Timestamp
  deleted_at: Timestamp?
}

entity QuotationItem {
  id: UUID PK
  quotation_id: UUID FK→quotations
  line_no: i32
  product_id: UUID FK→products
  description: String
  quantity: Decimal(10,6)
  unit: String
  unit_price: Decimal(10,6)
  unit_cost: Decimal(10,6) // BOM展开成本
  discount_rate: Decimal(5,2)
  amount: Decimal(10,6)
  delivery_date: Date?
}

enum QuotationStatus {
  DRAFT, SENT, ACCEPTED, REJECTED, EXPIRED
  }

共享层交互：DocumentSequence 生成编号 | CostEntry 无直接写入（报价阶段不产生实际成本）

SalesOrder 销售订单

功能：多方线上会签确认，合同风险控制，全流程起点

entity SalesOrder {
  id: UUID PK
  doc_number: String // SO-2026-05-xxxxx
  customer_id: UUID FK→customers
  contact_id: UUID FK→contacts
  order_date: Date
  status: SalesOrderStatus
  total_amount: Decimal(10,6)
  payment_terms: String
  delivery_address: String
  remark: String
  operator_id: UUID FK→users
  created_at: Timestamp
  updated_at: Timestamp
  deleted_at: Timestamp?
}

entity SalesOrderItem {
  id: UUID PK
  order_id: UUID FK→sales_orders
  line_no: i32
  product_id: UUID FK→products
  description: String
  quantity: Decimal(10,6)
  unit_price: Decimal(10,6)
  amount: Decimal(10,6)
  delivered_qty: Decimal(10,6) // 已发货量
  returned_qty: Decimal(10,6) // 已退货量
  delivery_date: Date?
}

enum SalesOrderStatus {
  DRAFT, CONFIRMED, IN_PRODUCTION
  PARTIALLY_SHIPPED, SHIPPED, COMPLETED
  CANCELLED
  }

共享层交互：DocumentSequence 编号 | InventoryReservation SOFT预留 | DocumentLink 关联报价

ShippingRequest 发货申请

功能：销售发货指令枪，直接流转至仓库触发出库

entity ShippingRequest {
  id: UUID PK
  doc_number: String // SR-2026-05-xxxxx
  order_id: UUID FK→sales_orders
  customer_id: UUID FK→customers
  request_date: Date
  expected_ship_date: Date
  status: ShippingStatus
  shipping_address: String
  carrier: String? // 物流商
  tracking_number: String?
  remark: String
  operator_id: UUID FK→users
  created_at: Timestamp
  updated_at: Timestamp
  deleted_at: Timestamp?
}

entity ShippingRequestItem {
  id: UUID PK
  request_id: UUID FK→shipping_requests
  order_item_id: UUID FK→sales_order_items
  product_id: UUID FK→products
  requested_qty: Decimal(10,6)
  shipped_qty: Decimal(10,6) // 实际出库量
  warehouse_id: UUID FK→warehouses
}

enum ShippingStatus {
  DRAFT, CONFIRMED, PICKING, SHIPPED
  CANCELLED
  }

共享层交互：InventoryReservation 释放预留 | DocumentLink 关联订单 | CostEntry 记销售成本

SalesReturn 销售退货

功能：逆向退货处理，触发质量追溯和库存回库

entity SalesReturn {
  id: UUID PK
  doc_number: String // SRT-2026-05-xxxxx
  order_id: UUID FK→sales_orders
  shipping_request_id: UUID FK→shipping_requests
  customer_id: UUID FK→customers
  return_date: Date
  status: ReturnStatus
  return_reason: String
  total_amount: Decimal(10,6)
  remark: String
  operator_id: UUID FK→users
  created_at: Timestamp
  updated_at: Timestamp
  deleted_at: Timestamp?
}

entity SalesReturnItem {
  id: UUID PK
  return_id: UUID FK→sales_returns
  order_item_id: UUID FK→sales_order_items
  product_id: UUID FK→products
  returned_qty: Decimal(10,6)
  unit_price: Decimal(10,6)
  amount: Decimal(10,6)
  disposition: ReturnDisposition
}

enum ReturnStatus {
  DRAFT, CONFIRMED, RECEIVED, INSPECTING
  COMPLETED, CANCELLED
  }
enum ReturnDisposition {
  RESTOCK, SCRAP, REWORK
  }

共享层交互：DocumentLink 关联原发货 | CostEntry 冲减收入 | 可触发 QMS 检验

Reconciliation 月对账单

功能：月底一键导出客户发货明细，核对后自动生成财务应收账款，打通业财融合

entity Reconciliation {
  id: UUID PK
  doc_number: String // REC-2026-05-xxxxx
  customer_id: UUID FK→customers
  period: String // "2026-05"
  status: ReconciliationStatus
  total_amount: Decimal(10,6)
  confirmed_amount: Decimal(10,6)
  difference: Decimal(10,6) // 差异自动标记
  remark: String
  operator_id: UUID FK→users
  created_at: Timestamp
  updated_at: Timestamp
  deleted_at: Timestamp?
}

entity ReconciliationItem {
  id: UUID PK
  reconciliation_id: UUID FK→reconciliations
  shipping_request_id: UUID FK→shipping_requests
  product_id: UUID FK→products
  quantity: Decimal(10,6)
  unit_price: Decimal(10,6)
  amount: Decimal(10,6)
  confirmed: bool
  remark: String?
}

ReconciliationStatus

enum ReconciliationStatus {
  DRAFT, SENT, CONFIRMED, DISPUTED
  SETTLED
  }

共享层交互

DocumentSequence 编号
DocumentLink 关联发货单（RECONCILES）
CostEntry 对账确认后生成应收账款凭证
（后续 FMS 模块消费此凭证进行核销）
